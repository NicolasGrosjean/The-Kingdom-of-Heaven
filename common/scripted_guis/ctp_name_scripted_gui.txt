random_saint_name = {
	scope = character
	effect = {
		hidden_effect = {
			faith = {
				random_in_list = {
					variable = faith_saints
					limit = { is_male = yes }
					save_temporary_scope_as = named_saint
				}
			}
			change_first_name = scope:named_saint
		}
	}
}

faith_has_male_saints = {
	scope = character
	is_shown = {
		faith = {
			has_variable_list = faith_saints
			any_in_list = {
				variable = faith_saints
				limit = { is_male = yes }
				exists = this
			}
		}
	}
}
